-- Kích hoạt ràng buộc khóa ngoại
PRAGMA foreign_keys = ON;

/* =======================================================================
   1. CORE & UNIFIED IMAGES 
   ======================================================================= */

CREATE TABLE Users (
    UserId       INTEGER PRIMARY KEY AUTOINCREMENT,
    PhoneNumber  TEXT UNIQUE NOT NULL,
    Email        TEXT UNIQUE NULL,
    PasswordHash BLOB NOT NULL,
    RoleType     TEXT NOT NULL CHECK (RoleType IN ('FARMER','SME','ADMIN')),
    DisplayName  TEXT NULL,
    IsActive     INTEGER NOT NULL DEFAULT 1,
    CreatedAt    TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Images (
    ImageId       INTEGER PRIMARY KEY AUTOINCREMENT,
    ReferenceId   INTEGER NOT NULL,
    -- Đã bổ sung 'MACHINE' vào danh sách
    ReferenceType TEXT NOT NULL CHECK (ReferenceType IN ('USER', 'PRODUCT', 'SME', 'FARM', 'CONSULTING', 'MACHINE')),
    ImageUrl      TEXT NOT NULL,
    IsPrimary     INTEGER DEFAULT 0,
    DisplayOrder  INTEGER DEFAULT 0,
    UploadedAt    TEXT DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IX_Images_Reference ON Images(ReferenceType, ReferenceId);

CREATE TABLE core_UserAddresses (
    AddressId   INTEGER PRIMARY KEY AUTOINCREMENT,
    UserId      INTEGER NOT NULL,
    Province    TEXT,
    District    TEXT,
    Commune     TEXT,
    AddressLine TEXT,
    CreatedAt   TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

CREATE TABLE FarmerProfiles (
    UserId         INTEGER PRIMARY KEY, 
    FullName       TEXT NOT NULL,
    Village        TEXT NULL,
    ContactName    TEXT NULL,
    ContactPhone   TEXT NULL,
    PreferredVoice INTEGER DEFAULT 1,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

CREATE TABLE SMEProfiles (
    UserId         INTEGER PRIMARY KEY, 
    CompanyName    TEXT NOT NULL,
    TaxCode        TEXT UNIQUE NOT NULL,
    ContactName    TEXT NULL,
    ContactPhone   TEXT NULL,
    AddressSummary TEXT NULL,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

/* =======================================================================
   2. IOT & NÔNG NGHIỆP
   ======================================================================= */

CREATE TABLE iot_Farms (
    FarmId         INTEGER PRIMARY KEY AUTOINCREMENT,
    FarmerId       INTEGER NOT NULL,
    FarmName       TEXT,
    Location       TEXT NULL, 
    AreaHectares   REAL NOT NULL,
    CropType       TEXT,
    Certifications TEXT NULL,
    FOREIGN KEY (FarmerId) REFERENCES FarmerProfiles(UserId)
);

CREATE TABLE iot_IoTDevices (
    DeviceId     INTEGER PRIMARY KEY AUTOINCREMENT,
    FarmId       INTEGER NOT NULL,
    DeviceType   TEXT CHECK (DeviceType IN ('SOIL','AIR','WATER','GATEWAY')),
    Description  TEXT NULL, -- BỔ SUNG MÔ TẢ THIẾT BỊ
    BatteryLevel INTEGER CHECK (BatteryLevel BETWEEN 0 AND 100),
    FOREIGN KEY (FarmId) REFERENCES iot_Farms(FarmId)
);

CREATE TABLE iot_IoTSensorReadings (
    ReadingId   INTEGER PRIMARY KEY AUTOINCREMENT,
    DeviceId    INTEGER NOT NULL,
    MetricType  TEXT NOT NULL,
    MetricValue REAL NOT NULL,
    RecordedAt  TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (DeviceId) REFERENCES iot_IoTDevices(DeviceId)
);

CREATE TABLE iot_FarmingLogs (
    LogId        INTEGER PRIMARY KEY AUTOINCREMENT,
    FarmId       INTEGER NOT NULL,
    ActionType   TEXT NOT NULL,
    MaterialUsed TEXT NULL,
    Quantity     REAL NULL,
    RecordedAt   TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (FarmId) REFERENCES iot_Farms(FarmId)
);

/* =======================================================================
   3. LOGISTICS (Thuê máy móc)
   ======================================================================= */

CREATE TABLE logistics_AgriMachines (
    MachineId        INTEGER PRIMARY KEY AUTOINCREMENT,
    OwnerId          INTEGER NOT NULL,
    MachineType      TEXT NOT NULL,
    Description      TEXT NULL, -- BỔ SUNG MÔ TẢ MÁY MÓC
    BasePricePerHour REAL NOT NULL,
    FOREIGN KEY (OwnerId) REFERENCES Users(UserId)
);

CREATE TABLE logistics_MachineHailingRequests (
    RequestId           INTEGER PRIMARY KEY AUTOINCREMENT,
    FarmId              INTEGER NOT NULL,
    MachineTypeRequired TEXT NOT NULL,
    ExpectedStartTime   TEXT NOT NULL,
    Status              TEXT DEFAULT 'MATCHING',
    FOREIGN KEY (FarmId) REFERENCES iot_Farms(FarmId)
);

/* =======================================================================
   4. COMMERCE (Sản phẩm, Đơn hàng, Đất đai)
   ======================================================================= */

CREATE TABLE commerce_Products (
    ProductId   INTEGER PRIMARY KEY AUTOINCREMENT,
    SellerId    INTEGER NOT NULL,
    Title       TEXT NOT NULL,
    Description TEXT NULL, -- SẼ ĐƯỢC ĐIỀN ĐẦY ĐỦ Ở PHẦN INSERT
    Category    TEXT NULL,
    Price       REAL NOT NULL,
    Unit        TEXT NULL,
    CreatedAt   TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (SellerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_ProductBatches (
    ProductBatchId INTEGER PRIMARY KEY AUTOINCREMENT,
    ProductId      INTEGER NOT NULL,
    BatchCode      TEXT UNIQUE,
    ProductionDate TEXT NULL,
    FOREIGN KEY (ProductId) REFERENCES commerce_Products(ProductId)
);

CREATE TABLE commerce_Orders (
    OrderId    INTEGER PRIMARY KEY AUTOINCREMENT,
    BuyerId    INTEGER NOT NULL,
    OrderTotal REAL NOT NULL,
    Status     TEXT DEFAULT 'CREATED',
    CreatedAt  TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (BuyerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_OrderItems (
    OrderItemId INTEGER PRIMARY KEY AUTOINCREMENT,
    OrderId     INTEGER NOT NULL,
    ProductId   INTEGER NOT NULL,
    Quantity    REAL NOT NULL,
    Price       REAL NOT NULL,
    FOREIGN KEY (OrderId) REFERENCES commerce_Orders(OrderId),
    FOREIGN KEY (ProductId) REFERENCES commerce_Products(ProductId)
);

CREATE TABLE commerce_LandListings (
    LandId          INTEGER PRIMARY KEY AUTOINCREMENT,
    OwnerId         INTEGER NOT NULL,
    AreaHectares    REAL NOT NULL,
    PricePerHectare REAL NULL,
    FOREIGN KEY (OwnerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_LandRentals (
    RentalId  INTEGER PRIMARY KEY AUTOINCREMENT,
    LandId    INTEGER NOT NULL,
    RenterId  INTEGER NOT NULL,
    StartDate TEXT, 
    FOREIGN KEY (LandId) REFERENCES commerce_LandListings(LandId),
    FOREIGN KEY (RenterId) REFERENCES Users(UserId)
);

/* =======================================================================
   5. COMMUNITY (Tư vấn, Thông báo)
   ======================================================================= */

CREATE TABLE community_ConsultingCases (
    CaseId   INTEGER PRIMARY KEY AUTOINCREMENT,
    FarmerId INTEGER NOT NULL,
    Title    TEXT,
    Status   TEXT DEFAULT 'OPEN',
    FOREIGN KEY (FarmerId) REFERENCES FarmerProfiles(UserId)
);

CREATE TABLE Notifications (
    NotificationId INTEGER PRIMARY KEY AUTOINCREMENT,
    UserId         INTEGER NOT NULL,
    Title          TEXT,
    IsRead         INTEGER DEFAULT 0,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);
