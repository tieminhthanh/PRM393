-- Để bật Foreign Key (Rất quan trọng trong SQLite)
PRAGMA foreign_keys = ON;

/* =======================================================================
   1. CORE & UNIFIED IMAGES (Hình ảnh tập trung)
   ======================================================================= */

CREATE TABLE Users (
    UserId TEXT PRIMARY KEY,
    PhoneNumber TEXT UNIQUE NOT NULL,
    Email TEXT UNIQUE,
    PasswordHash BLOB NOT NULL,
    RoleType TEXT NOT NULL CHECK (RoleType IN ('FARMER','SME','ADMIN')),
    DisplayName TEXT,
    IsActive INTEGER NOT NULL DEFAULT 1,
    CreatedAt TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE Images (
    ImageId TEXT PRIMARY KEY,
    ReferenceId TEXT NOT NULL, 
    ReferenceType TEXT NOT NULL CHECK (ReferenceType IN ('USER', 'PRODUCT', 'SME', 'FARM', 'CONSULTING')),
    ImageUrl TEXT NOT NULL,
    IsPrimary INTEGER DEFAULT 0, 
    DisplayOrder INTEGER DEFAULT 0,
    UploadedAt TEXT DEFAULT CURRENT_TIMESTAMP
);
CREATE INDEX IX_Images_Reference ON Images(ReferenceType, ReferenceId);

CREATE TABLE core_UserAddresses (
    AddressId TEXT PRIMARY KEY,
    UserId TEXT NOT NULL,
    Province TEXT,
    District TEXT,
    Commune TEXT,
    AddressLine TEXT,
    CreatedAt TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

CREATE TABLE FarmerProfiles (
    UserId TEXT PRIMARY KEY,
    FullName TEXT NOT NULL,
    Village TEXT,
    ContactName TEXT,
    ContactPhone TEXT,
    PreferredVoice INTEGER DEFAULT 1,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

CREATE TABLE SMEProfiles (
    UserId TEXT PRIMARY KEY,
    CompanyName TEXT NOT NULL,
    TaxCode TEXT UNIQUE NOT NULL,
    ContactName TEXT,
    ContactPhone TEXT,
    AddressSummary TEXT,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

/* =======================================================================
   2. IOT & NÔNG NGHIỆP
   ======================================================================= */

CREATE TABLE iot_Farms (
    FarmId TEXT PRIMARY KEY,
    FarmerId TEXT NOT NULL,
    FarmName TEXT,
    Location TEXT, -- Lưu tọa độ dưới dạng TEXT hoặc WKT
    AreaHectares REAL NOT NULL,
    CropType TEXT,
    Certifications TEXT,
    FOREIGN KEY (FarmerId) REFERENCES FarmerProfiles(UserId)
);

CREATE TABLE iot_IoTDevices (
    DeviceId TEXT PRIMARY KEY,
    FarmId TEXT NOT NULL,
    DeviceType TEXT CHECK (DeviceType IN ('SOIL','AIR','WATER','GATEWAY')),
    BatteryLevel INTEGER CHECK (BatteryLevel BETWEEN 0 AND 100),
    FOREIGN KEY (FarmId) REFERENCES iot_Farms(FarmId)
);

CREATE TABLE iot_IoTSensorReadings (
    ReadingId INTEGER PRIMARY KEY AUTOINCREMENT,
    DeviceId TEXT NOT NULL,
    MetricType TEXT NOT NULL,
    MetricValue REAL NOT NULL,
    RecordedAt TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (DeviceId) REFERENCES iot_IoTDevices(DeviceId)
);

CREATE TABLE iot_FarmingLogs (
    LogId TEXT PRIMARY KEY,
    FarmId TEXT NOT NULL,
    ActionType TEXT NOT NULL,
    MaterialUsed TEXT,
    Quantity REAL,
    RecordedAt TEXT NOT NULL DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (FarmId) REFERENCES iot_Farms(FarmId)
);

/* =======================================================================
   3. LOGISTICS (Thuê máy móc)
   ======================================================================= */

CREATE TABLE logistics_AgriMachines (
    MachineId TEXT PRIMARY KEY,
    OwnerId TEXT NOT NULL,
    MachineType TEXT NOT NULL,
    BasePricePerHour REAL NOT NULL,
    FOREIGN KEY (OwnerId) REFERENCES Users(UserId)
);

CREATE TABLE logistics_MachineHailingRequests (
    RequestId TEXT PRIMARY KEY,
    FarmId TEXT NOT NULL,
    MachineTypeRequired TEXT NOT NULL,
    ExpectedStartTime TEXT NOT NULL,
    Status TEXT DEFAULT 'MATCHING',
    FOREIGN KEY (FarmId) REFERENCES iot_Farms(FarmId)
);

/* =======================================================================
   4. COMMERCE (Sản phẩm, Giỏ hàng, Đơn hàng)
   ======================================================================= */

CREATE TABLE commerce_Products (
    ProductId TEXT PRIMARY KEY,
    SellerId TEXT NOT NULL,
    Title TEXT NOT NULL,
    Description TEXT,
    Category TEXT,
    Price REAL NOT NULL,
    Unit TEXT,
    CreatedAt TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (SellerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_ProductBatches (
    ProductBatchId TEXT PRIMARY KEY,
    ProductId TEXT NOT NULL,
    BatchCode TEXT UNIQUE,
    ProductionDate TEXT,
    FOREIGN KEY (ProductId) REFERENCES commerce_Products(ProductId)
);

CREATE TABLE commerce_Carts (
    CartId TEXT PRIMARY KEY,
    UserId TEXT NOT NULL,
    UpdatedAt TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_CartItems (
    CartItemId TEXT PRIMARY KEY,
    CartId TEXT NOT NULL,
    ProductId TEXT NOT NULL,
    Quantity REAL NOT NULL,
    FOREIGN KEY (CartId) REFERENCES commerce_Carts(CartId),
    FOREIGN KEY (ProductId) REFERENCES commerce_Products(ProductId)
);

CREATE TABLE commerce_Orders (
    OrderId TEXT PRIMARY KEY,
    BuyerId TEXT NOT NULL,
    OrderTotal REAL NOT NULL,
    Status TEXT DEFAULT 'CREATED',
    CreatedAt TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (BuyerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_OrderItems (
    OrderItemId TEXT PRIMARY KEY,
    OrderId TEXT NOT NULL,
    ProductId TEXT NOT NULL,
    Quantity REAL NOT NULL,
    Price REAL NOT NULL,
    FOREIGN KEY (OrderId) REFERENCES commerce_Orders(OrderId),
    FOREIGN KEY (ProductId) REFERENCES commerce_Products(ProductId)
);

CREATE TABLE commerce_Payments (
    PaymentId TEXT PRIMARY KEY,
    OrderId TEXT NOT NULL,
    Amount REAL NOT NULL,
    Provider TEXT,
    PaidAt TEXT DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (OrderId) REFERENCES commerce_Orders(OrderId)
);

CREATE TABLE commerce_Certificates (
    CertificateId TEXT PRIMARY KEY,
    SellerId TEXT NOT NULL,
    CertificateType TEXT,
    FOREIGN KEY (SellerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_LandListings (
    LandId TEXT PRIMARY KEY,
    OwnerId TEXT NOT NULL,
    AreaHectares REAL NOT NULL,
    PricePerHectare REAL,
    FOREIGN KEY (OwnerId) REFERENCES Users(UserId)
);

CREATE TABLE commerce_LandRentals (
    RentalId TEXT PRIMARY KEY,
    LandId TEXT NOT NULL,
    RenterId TEXT NOT NULL,
    StartDate TEXT,
    FOREIGN KEY (LandId) REFERENCES commerce_LandListings(LandId),
    FOREIGN KEY (RenterId) REFERENCES Users(UserId)
);

/* =======================================================================
   5. COMMUNITY & SYSTEM
   ======================================================================= */

CREATE TABLE community_ConsultingCases (
    CaseId TEXT PRIMARY KEY,
    FarmerId TEXT NOT NULL,
    Title TEXT,
    Status TEXT DEFAULT 'OPEN',
    FOREIGN KEY (FarmerId) REFERENCES FarmerProfiles(UserId)
);

CREATE TABLE NotificationChannels (
    ChannelId INTEGER PRIMARY KEY AUTOINCREMENT,
    Name TEXT
);

CREATE TABLE Notifications (
    NotificationId TEXT PRIMARY KEY,
    UserId TEXT NOT NULL,
    Title TEXT,
    IsRead INTEGER DEFAULT 0,
    FOREIGN KEY (UserId) REFERENCES Users(UserId)
);

CREATE TABLE AuditLogs (
    AuditId INTEGER PRIMARY KEY AUTOINCREMENT,
    Action TEXT,
    Details TEXT
);

-- =======================================================================
-- TRUY VẤN MẪU TEST (Đã được chuẩn hóa cho SQLite)
-- =======================================================================

SELECT 
    p.Title AS [Tên Sản Phẩm],
    p.Category AS [Danh Mục],
    p.Price AS [Đơn Giá (VNĐ)],
    ci.Quantity AS [Số Lượng Trong Giỏ],
    (p.Price * ci.Quantity) AS [Thành Tiền (VNĐ)],
    i.ImageUrl AS [Link Ảnh Đại Diện]
FROM commerce_Carts c
INNER JOIN commerce_CartItems ci 
    ON c.CartId = ci.CartId
INNER JOIN commerce_Products p 
    ON ci.ProductId = p.ProductId
LEFT JOIN Images i 
    ON p.ProductId = i.ReferenceId 
    AND i.ReferenceType = 'PRODUCT' 
    AND i.IsPrimary = 1
WHERE c.UserId = '11111111-1111-1111-1111-111111111111';
